21 void ReportQueue::run() {
22  // Check if queue is nonempty. If so, move any reports to the Json array and
23  // try to send it to the server. Clear the Json array only if the send
24  // succeeds.
25  Json::Value report_array(Json::arrayValue);
26  while (!shutdown_) {
27  std::unique_lock<std::mutex> thread_lock(thread_mutex_);
28  if (!report_queue_.empty()) {
29  std::lock_guard<std::mutex> queue_lock(queue_mutex_);
30  while (!report_queue_.empty()) {
31  report_array.append(*report_queue_.front());
32  report_queue_.pop();
33  }
34  }
35  if (report_array.size() > 0) {
36  HttpResponse response = http->post(config.tls.server + "/events", report_array);
37  // 404 implies the server does not support this feature. Nothing we can
38  // do, just move along.
39  if (response.isOk() || response.http_status_code == 404) {
40  report_array = Json::arrayValue;
41  }
42  }
43  cv_.wait_for(thread_lock, std::chrono::seconds(10));
44  }
45 }
