59 void Aktualizr::CycleEventHandler::handle(const std::shared_ptr<event::BaseEvent> &event) {
60  if (!running) {
61  return;
62  }
63 
64  RunningMode running_mode = aktualizr.config_.uptane.running_mode;
65 
66  if (event->variant == "FetchMetaComplete") {
67  aktualizr.CheckUpdates();
68  } else if (event->variant == "NoUpdateAvailable") {
69  breakLoop();
70  } else if (event->variant == "UpdateAvailable") {
71  auto ua_event = dynamic_cast<event::UpdateAvailable *>(event.get());
72  assert(!ua_event->updates.empty());
73  if (running_mode == RunningMode::kCheck) {
74  breakLoop();
75  } else if (running_mode == RunningMode::kInstall) {
76  aktualizr.Install(ua_event->updates);
77  } else {
78  aktualizr.Download(ua_event->updates);
79  }
80  } else if (event->variant == "NothingToDownload") {
81  breakLoop();
82  } else if (event->variant == "DownloadProgressReport") {
83  // silent
84  } else if (event->variant == "AllDownloadsComplete") {
85  auto dc_event = dynamic_cast<event::AllDownloadsComplete *>(event.get());
86  if (running_mode == RunningMode::kDownload) {
87  breakLoop();
88  } else {
89  aktualizr.Install(dc_event->updates);
90  }
91  } else if (event->variant == "AllInstallsComplete") {
92  // finished installing everything
93  aktualizr.uptane_client_->putManifest();
94  } else if (event->variant == "PutManifestComplete") {
95  boost::filesystem::path reboot_flag = "/tmp/aktualizr_reboot_flag";
96  if (boost::filesystem::exists(reboot_flag)) {
97  boost::filesystem::remove(reboot_flag);
98  if (getppid() == 1) { // if parent process id is 1, aktualizr runs under systemd
99  exit(0); // aktualizr service runs with 'Restart=always' systemd option.
100  // Systemd will start aktualizr automatically if it exit.
101  } else { // Aktualizr runs from terminal
102  LOG_INFO << "Aktualizr has been updated and requires restarting to run the new version.";
103  }
104  }
105  breakLoop();
106  } else if (event->variant == "Error") {
107  auto err_event = dynamic_cast<event::Error *>(event.get());
108  LOG_WARNING << "got Error event: " << err_event->message;
109  breakLoop();
110  }
111 }
