38  std::uniform_int_distribution<uint8_t> random_byte(0x00, 0xFF);
39 
40  std::vector<unsigned char> block(kUpdateImageBlockSize);
41  std::ofstream imagefile(kImageFilePath.c_str(), std::ios::binary | std::ios::app);
42  MultiPartSHA256Hasher hasher;
43  std::size_t size = 0;
44  while (size < kUpdateImageSize) {
45  std::generate(block.begin(), block.end(), std::bind(random_byte, std::ref(gen)));
46  hasher.update(&block[0], block.size());
47  std::copy(block.begin(), block.end(), std::ostreambuf_iterator<char>(imagefile.rdbuf()));
48  size += std::min(block.size(), kUpdateImageSize - size);
49  }
