58 void Aktualizr::UptaneCycle() {
59  result::UpdateCheck update_result = CheckUpdates().get();
60  if (update_result.updates.size() == 0) {
61  return;
62  }
63 
64  result::Download download_result = Download(update_result.updates).get();
65  if (download_result.status != result::DownloadStatus::kSuccess || download_result.updates.size() == 0) {
66  return;
67  }
68 
69  uptane_client_->uptaneInstall(download_result.updates);
70  uptane_client_->putManifest();
71  if (uptane_client_->isInstallCompletionRequired()) {
72  // If there are some pending updates then effectively either reboot (ostree) or aktualizr restart (fake pack mngr)
73  // is required to apply the update(s)
74  LOG_INFO << "About to exit aktualizr so the pending updates can be applied after reboot";
75  Shutdown();
76  }
77 }
