169 void Config::postUpdateValues() {
170  logger_set_threshold(logger);
171 
172  if (provision.provision_path.empty()) {
173  provision.mode = ProvisionMode::kImplicit;
174  }
175 
176  if (tls.server.empty()) {
177  if (!tls.server_url_path.empty()) {
178  try {
179  tls.server = Utils::readFile(tls.server_url_path);
180  } catch (const boost::filesystem::filesystem_error& e) {
181  LOG_ERROR << "Couldn't read gateway URL: " << e.what();
182  tls.server = "";
183  }
184  } else if (!provision.provision_path.empty()) {
185  if (boost::filesystem::exists(provision.provision_path)) {
186  tls.server = Bootstrap::readServerUrl(provision.provision_path);
187  } else {
188  LOG_ERROR << "Provided provision archive " << provision.provision_path << " does not exist!";
189  }
190  }
191  }
192 
193  if (!tls.server.empty()) {
194  if (provision.server.empty()) {
195  provision.server = tls.server;
196  }
197 
198  if (uptane.repo_server.empty()) {
199  uptane.repo_server = tls.server + "/repo";
200  }
201 
202  if (uptane.director_server.empty()) {
203  uptane.director_server = tls.server + "/director";
204  }
205 
206  if (pacman.ostree_server.empty()) {
207  pacman.ostree_server = tls.server + "/treehub";
208  }
209  }
210 
211  if (!uptane.director_server.empty()) {
212  if (provision.ecu_registration_endpoint.empty()) {
213  provision.ecu_registration_endpoint = uptane.director_server + "/ecus";
214  }

292 void Config::checkLegacyVersion() {
293  if (uptane.legacy_interface.empty()) {
294  return;
295  }
296  if (!boost::filesystem::exists(uptane.legacy_interface)) {
297  throw FatalException(std::string("Legacy external flasher not found: ") + uptane.legacy_interface.string());
298  }
299  std::stringstream command;
300  std::string output;
301  command << uptane.legacy_interface << " api-version --loglevel " << loggerGetSeverity();
302  int rs = Utils::shell(command.str(), &output);
303  if (rs != 0) {
304  throw FatalException(std::string("Legacy external flasher api-version command failed: ") + output);
305  }
306  boost::trim_if(output, boost::is_any_of(" \n\r\t"));
307  if (output != "1") {
308  throw FatalException(std::string("Unexpected legacy external flasher API version: ") + output);
309  }

312 void Config::initLegacySecondaries() {
313  if (uptane.legacy_interface.empty()) {
314  return;
315  }
316  std::stringstream command;
317  std::string output;
318  command << uptane.legacy_interface << " list-ecus --loglevel " << loggerGetSeverity();
319  int rs = Utils::shell(command.str(), &output);
320  if (rs != 0) {
321  LOG_ERROR << "Legacy external flasher list-ecus command failed: " << output;
322  return;
323  }
324 
325  std::stringstream ss(output);
326  std::string buffer;
327  while (std::getline(ss, buffer, '\n')) {
328  Uptane::SecondaryConfig sconfig;
329  sconfig.secondary_type = Uptane::SecondaryType::kLegacy;
330  std::vector<std::string> ecu_info;
331  boost::split(ecu_info, buffer, boost::is_any_of(" \n\r\t"), boost::token_compress_on);
332  if (ecu_info.size() == 0) {
333  // Could print a warning but why bother.
334  continue;
335  }
336  if (ecu_info.size() == 1) {
337  sconfig.ecu_hardware_id = ecu_info[0];
338  // Use getSerial, which will get the public_key_id, initialized in ManagedSecondary constructor.
339  sconfig.ecu_serial = "";
340  sconfig.full_client_dir = storage.path / sconfig.ecu_hardware_id;
341  LOG_INFO << "Legacy ECU configured with hardware ID " << sconfig.ecu_hardware_id;
342  } else if (ecu_info.size() >= 2) {
343  // For now, silently ignore anything after the second token.
344  sconfig.ecu_hardware_id = ecu_info[0];
345  sconfig.ecu_serial = ecu_info[1];
346  sconfig.full_client_dir = storage.path / (sconfig.ecu_hardware_id + "-" + sconfig.ecu_serial);
347  LOG_INFO << "Legacy ECU configured with hardware ID " << sconfig.ecu_hardware_id << " and serial "
348  << sconfig.ecu_serial;
349  }
350 
351  sconfig.partial_verifying = false;
352  sconfig.ecu_private_key = "sec.private";
353  sconfig.ecu_public_key = "sec.public";
354 
355  sconfig.firmware_path = sconfig.full_client_dir / "firmware.bin";
356  sconfig.metadata_path = sconfig.full_client_dir / "metadata";
357  sconfig.target_name_path = sconfig.full_client_dir / "target_name";
358  sconfig.flasher = uptane.legacy_interface;
359 
360  uptane.secondary_configs.push_back(sconfig);
361  }
362 }
