50  recursive_check = [&keys, &recursive_check](const boost::filesystem::path &path) {
51  for (auto &p : boost::filesystem::directory_iterator(path)) {
52  std::cout << "PATH: " << p.path() << "\n";
53  if (p.status().type() == boost::filesystem::file_type::directory_file) {
54  recursive_check(p.path());
55  } else {
56  if (p.path().filename().string() == "key_type") {
57  continue;
58  }
59  std::string hash = Crypto::sha512digest(Utils::readFile(p.path()));
60  if (std::find(keys.begin(), keys.end(), hash) == keys.end()) {
61  keys.push_back(hash);

63  FAIL() << p.path().string() << " is not unique";

65  }
